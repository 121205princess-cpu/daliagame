using System; using System.Collections.Generic; public class Skill { public string Name; public double Cooldown; public DateTime NextAvailable = DateTime.MinValue; public Skill(string name, double cooldown) { Name = name; Cooldown = cooldown; } } public class Weapon { public string Name; public Skill ESkill; public Skill QSkill; public Weapon(string name, string eName, double eCd, string qName, double qCd) { Name = name; ESkill = new Skill(eName, eCd); QSkill = new Skill(qName, qCd); } } public class Program { static List weapons = new List { new Weapon("성검", "돌진 베기", 5.0, "심판의 낙뢰", 15.0), new Weapon("마법 총", "속사 모드", 8.0, "마력 폭격", 20.0) }; static int currentWeaponIdx = 0; public static void Main() { while (true) { var current = weapons[currentWeaponIdx]; Console.WriteLine($"[현재 무기: {current.Name}] (평타, E스킬, 궁극기, 무기교체, 종료)"); string input = Console.ReadLine(); if (input == "종료") break; if (input == "평타") { int dmg = new Random().Next(10, 20); Console.WriteLine($"✨ 핑크색 반짝임과 함께 {dmg} 데미지! (효과음: 챙!)"); } else if (input == "E스킬" || input == "궁극기") { Skill target = (input == "E스킬") ? current.ESkill : current.QSkill; UseSkill(target); } else if (input == "무기교체") { currentWeaponIdx = (currentWeaponIdx == 1) ? 0 : 1; Console.WriteLine($"🔄 무기가 {weapons[currentWeaponIdx].Name}으로 변경되었습니다."); } } } static void UseSkill(Skill s) { if (DateTime.Now >= s.NextAvailable) { Console.WriteLine($"🔥 {s.Name} 발동!!"); s.NextAvailable = DateTime.Now.AddSeconds(s.Cooldown); } else { Console.WriteLine($"❌ {s.Name} 쿨타임 중... ({Math.Ceiling((s.NextAvailable - DateTime.Now).TotalSeconds)}초)"); } } }
