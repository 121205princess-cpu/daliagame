class GameEngine { constructor() { this.state = { wave: 0, currentWeapon: '칼', isBossWave: false, shakeIntensity: 0, cooldowns: { eSkill: 0, ultimate: 0 } }; } // 웨이브 진입 시 상태 업데이트 (불안정성 강화) nextWave() { this.state.wave++; this.state.shakeIntensity = this.state.wave * 2; // 웨이브 높을수록 흔들림 증가 console.log(`웨이브 ${this.state.wave} 시작! 화면 불안정 레벨: ${this.state.shakeIntensity}`); if (this.state.wave === 3) { this.state.isBossWave = true; console.log("⚠️ 기괴한 보스 등장!"); } } // 무기/스킬 로직 (상태 기반 관리) performAction(action) { switch(action) { case '평타': return `⚔️ ${this.state.currentWeapon} 공격! [핑크색 반짝이 이펙트] 발생. 데미지: ${Math.floor(Math.random() * 50) + 50}`; case 'E스킬': if (Date.now() < this.state.cooldowns.eSkill) return "❌ 쿨타임 중입니다."; this.state.cooldowns.eSkill = Date.now() + 5000; return this.state.currentWeapon === '칼' ? "⚡ 돌격!" : "🔫 원거리 사격!"; case '무기교체': this.state.currentWeapon = (this.state.currentWeapon === '칼') ? '총' : '칼'; return `🔄 무기 변경됨: ${this.state.currentWeapon}`; } } } // 사용 예시 const game = new GameEngine(); game.nextWave(); console.log(game.performAction('평타')); console.log(game.performAction('무기교체'));
